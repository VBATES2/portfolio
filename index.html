Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 


# 3rd Course Challenge written by VICTOR BATES II
# This program follows the instructions from the
# Firehose Intro course at the Firehose Project.
# FOOBAR 

puts '                                    '

puts "  #########################################"
puts '  #                                       #'
puts '  #       LETS PLAY THE FOOBAR GAME       #'
puts '  #     ENTER A NUMBER AND LETS COUNT     #'
puts '  #       THE FOOs,BARs, & FOOBARs        #'
puts '  #         FROM 0 TO YOUR NUMBER         #'
puts '  #                                       #'
puts '  #########################################'

puts '                                    '

puts '-------------------------------------------'

puts '                                    '

player_number=gets.chomp.to_i
foobar=0
foo=0
bar=0
count=1

puts '                                    '

puts '-------------------------------------------'

puts '                                    '

puts 0

while count<=player_number
  
 if count%3 == 0 && count%5 == 0
   
   
   puts 'FOOBAR'
   foobar=foobar+1
 
      
   elsif count%3 == 0
   
   puts 'FOO'
   foo=foo+1
   
   elsif count%5 == 0
   
    puts 'BAR'
   
   bar=bar+1
   
   else
   
     puts count
      
 end
  
  count=count+1
  
end

puts '                                    '

puts '-------------------------------------------'

puts '                                    '

puts " There are #{foo} FOOS"

puts '                                    '

puts '-------------------------------------------'

puts '                                    '

puts " There are #{bar} BARS"

puts '                                    '

puts '-------------------------------------------'

puts '                                    '

puts " There are #{foobar} FOOBARS"

puts '                                    '

puts '-------------------------------------------'

puts '                                    '
